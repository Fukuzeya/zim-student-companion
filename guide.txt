# ğŸ“ Zim Student Companion - Production System Design

## Executive Summary

**Zim Student Companion (ZSC)** is a WhatsApp-first AI tutoring platform designed specifically for Zimbabwean students, leveraging RAG (Retrieval-Augmented Generation) to provide curriculum-aligned education support.

### Unique Value Propositions
1. **Socratic Learning Method** - Never gives direct answers; guides students step-by-step
2. **ZIMSEC-Aligned** - Grounded in official syllabi, past papers, and marking schemes
3. **Shona/Ndebele Support** - Local language explanations for complex concepts
4. **Offline-First Design** - Works with intermittent connectivity
5. **Parent Dashboard** - Real-time progress tracking and weekly reports
6. **Gamified Competitions** - School vs school, district leaderboards
7. **Adaptive Difficulty** - AI adjusts based on student performance

## 2. RAG System - Complete Deep Dive

### What is RAG? (For Beginners)

RAG (Retrieval-Augmented Generation) combines two powerful techniques:

1. **Retrieval**: Finding relevant documents from your knowledge base
2. **Generation**: Using an LLM to generate answers based on retrieved context

## 3. Database Schema Design

### PostgreSQL Schema

```sql
-- USERS & AUTHENTICATION
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    phone_number VARCHAR(20) UNIQUE NOT NULL,
    whatsapp_id VARCHAR(50) UNIQUE,
    role VARCHAR(20) NOT NULL CHECK (role IN ('student', 'parent', 'teacher', 'admin')),
    created_at TIMESTAMP DEFAULT NOW(),
    last_active TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    subscription_tier VARCHAR(20) DEFAULT 'free',
    subscription_expires_at TIMESTAMP
);

-- STUDENT PROFILES
CREATE TABLE students (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    school_name VARCHAR(200),
    district VARCHAR(100),
    province VARCHAR(100),
    education_level VARCHAR(20) NOT NULL CHECK (
        education_level IN ('primary', 'secondary', 'a_level')
    ),
    grade VARCHAR(20) NOT NULL, -- Grade 1-7, Form 1-6, Lower 6, Upper 6
    subjects JSONB DEFAULT '[]', -- List of enrolled subjects
    preferred_language VARCHAR(20) DEFAULT 'english', -- english, shona, ndebele
    daily_goal_minutes INTEGER DEFAULT 30,
    created_at TIMESTAMP DEFAULT NOW()
);

-- PARENT-STUDENT RELATIONSHIPS
CREATE TABLE parent_student_links (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    parent_user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    relationship VARCHAR(20), -- mother, father, guardian
    verified BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(parent_user_id, student_id)
);

-- SUBJECTS & CURRICULUM
CREATE TABLE subjects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    code VARCHAR(20) UNIQUE NOT NULL, -- ZIMSEC subject code
    education_level VARCHAR(20) NOT NULL,
    description TEXT,
    is_active BOOLEAN DEFAULT TRUE
);

-- TOPICS WITHIN SUBJECTS
CREATE TABLE topics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    subject_id UUID REFERENCES subjects(id) ON DELETE CASCADE,
    name VARCHAR(200) NOT NULL,
    description TEXT,
    grade VARCHAR(20) NOT NULL,
    syllabus_reference VARCHAR(100), -- Reference to ZIMSEC syllabus section
    order_index INTEGER,
    estimated_hours DECIMAL(4,1)
);

-- LEARNING OBJECTIVES
CREATE TABLE learning_objectives (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic_id UUID REFERENCES topics(id) ON DELETE CASCADE,
    description TEXT NOT NULL,
    bloom_level VARCHAR(20), -- remember, understand, apply, analyze, evaluate, create
    order_index INTEGER
);

-- QUESTIONS BANK
CREATE TABLE questions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic_id UUID REFERENCES topics(id),
    subject_id UUID REFERENCES subjects(id),
    question_text TEXT NOT NULL,
    question_type VARCHAR(30) NOT NULL CHECK (
        question_type IN ('multiple_choice', 'short_answer', 'long_answer', 
                         'calculation', 'diagram', 'essay', 'practical')
    ),
    options JSONB, -- For multiple choice
    correct_answer TEXT,
    marking_scheme TEXT,
    marks INTEGER,
    difficulty VARCHAR(20) CHECK (difficulty IN ('easy', 'medium', 'hard')),
    source VARCHAR(50), -- 'past_paper', 'generated', 'teacher_submitted'
    source_year INTEGER,
    source_paper VARCHAR(50), -- Paper 1, Paper 2, etc.
    times_attempted INTEGER DEFAULT 0,
    times_correct INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW(),
    vector_id VARCHAR(100) -- Reference to vector store
);

-- STUDENT PRACTICE SESSIONS
CREATE TABLE practice_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    subject_id UUID REFERENCES subjects(id),
    topic_id UUID REFERENCES topics(id),
    session_type VARCHAR(30) CHECK (
        session_type IN ('daily_practice', 'topic_practice', 'mock_exam', 
                        'competition', 'revision', 'homework')
    ),
    started_at TIMESTAMP DEFAULT NOW(),
    ended_at TIMESTAMP,
    total_questions INTEGER DEFAULT 0,
    correct_answers INTEGER DEFAULT 0,
    score_percentage DECIMAL(5,2),
    time_spent_seconds INTEGER,
    difficulty_level VARCHAR(20),
    status VARCHAR(20) DEFAULT 'in_progress'
);

-- INDIVIDUAL QUESTION ATTEMPTS
CREATE TABLE question_attempts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID REFERENCES practice_sessions(id) ON DELETE CASCADE,
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    question_id UUID REFERENCES questions(id),
    student_answer TEXT,
    is_correct BOOLEAN,
    marks_earned DECIMAL(4,1),
    marks_possible DECIMAL(4,1),
    time_spent_seconds INTEGER,
    hints_used INTEGER DEFAULT 0,
    ai_feedback TEXT,
    attempted_at TIMESTAMP DEFAULT NOW()
);

-- CONVERSATION HISTORY
CREATE TABLE conversations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    whatsapp_message_id VARCHAR(100),
    role VARCHAR(20) NOT NULL CHECK (role IN ('user', 'assistant', 'system')),
    content TEXT NOT NULL,
    context_type VARCHAR(30), -- 'question', 'explanation', 'practice', 'general'
    subject_id UUID REFERENCES subjects(id),
    topic_id UUID REFERENCES topics(id),
    tokens_used INTEGER,
    created_at TIMESTAMP DEFAULT NOW()
);

-- ACHIEVEMENTS & BADGES
CREATE TABLE achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    icon_url VARCHAR(500),
    points INTEGER DEFAULT 0,
    criteria JSONB, -- Conditions to earn
    achievement_type VARCHAR(30) -- 'streak', 'mastery', 'competition', 'milestone'
);

CREATE TABLE student_achievements (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    achievement_id UUID REFERENCES achievements(id),
    earned_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(student_id, achievement_id)
);

-- COMPETITIONS
CREATE TABLE competitions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(200) NOT NULL,
    description TEXT,
    subject_id UUID REFERENCES subjects(id),
    education_level VARCHAR(20),
    competition_type VARCHAR(30) CHECK (
        competition_type IN ('individual', 'school', 'district', 'national')
    ),
    start_date TIMESTAMP NOT NULL,
    end_date TIMESTAMP NOT NULL,
    max_participants INTEGER,
    prizes JSONB,
    status VARCHAR(20) DEFAULT 'upcoming',
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE competition_participants (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    competition_id UUID REFERENCES competitions(id) ON DELETE CASCADE,
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    score DECIMAL(10,2) DEFAULT 0,
    rank INTEGER,
    joined_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(competition_id, student_id)
);

-- STUDENT PROGRESS TRACKING
CREATE TABLE student_topic_progress (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    topic_id UUID REFERENCES topics(id) ON DELETE CASCADE,
    mastery_level DECIMAL(5,2) DEFAULT 0, -- 0-100
    questions_attempted INTEGER DEFAULT 0,
    questions_correct INTEGER DEFAULT 0,
    time_spent_minutes INTEGER DEFAULT 0,
    last_practiced TIMESTAMP,
    next_review_date DATE, -- Spaced repetition
    UNIQUE(student_id, topic_id)
);

-- DAILY STREAKS
CREATE TABLE student_streaks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    student_id UUID REFERENCES students(id) ON DELETE CASCADE UNIQUE,
    current_streak INTEGER DEFAULT 0,
    longest_streak INTEGER DEFAULT 0,
    last_activity_date DATE,
    total_active_days INTEGER DEFAULT 0
);

-- SUBSCRIPTIONS & PAYMENTS
CREATE TABLE subscription_plans (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    tier VARCHAR(20) NOT NULL, -- free, basic, premium, family
    price_usd DECIMAL(10,2),
    price_zwl DECIMAL(10,2),
    duration_days INTEGER,
    features JSONB,
    max_students INTEGER DEFAULT 1, -- For family plans
    is_active BOOLEAN DEFAULT TRUE
);

CREATE TABLE payments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    plan_id UUID REFERENCES subscription_plans(id),
    amount DECIMAL(10,2) NOT NULL,
    currency VARCHAR(10) NOT NULL,
    payment_method VARCHAR(30), -- ecocash, onemoney, innbucks, visa, bank
    payment_reference VARCHAR(100),
    status VARCHAR(20) DEFAULT 'pending',
    paynow_poll_url VARCHAR(500),
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

-- PARENT REPORTS
CREATE TABLE parent_reports (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    parent_user_id UUID REFERENCES users(id),
    student_id UUID REFERENCES students(id),
    report_type VARCHAR(30), -- daily, weekly, monthly
    report_data JSONB,
    generated_at TIMESTAMP DEFAULT NOW(),
    sent_at TIMESTAMP,
    delivery_method VARCHAR(20) -- whatsapp, email
);

-- NOTIFICATIONS
CREATE TABLE notifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    title VARCHAR(200),
    message TEXT,
    notification_type VARCHAR(30),
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW(),
    scheduled_for TIMESTAMP,
    sent_at TIMESTAMP
);

-- INDEXES FOR PERFORMANCE
CREATE INDEX idx_students_user ON students(user_id);
CREATE INDEX idx_conversations_student ON conversations(student_id, created_at DESC);
CREATE INDEX idx_question_attempts_student ON question_attempts(student_id, attempted_at DESC);
CREATE INDEX idx_practice_sessions_student ON practice_sessions(student_id, started_at DESC);
CREATE INDEX idx_questions_topic ON questions(topic_id);
CREATE INDEX idx_questions_difficulty ON questions(difficulty);
CREATE INDEX idx_student_progress_topic ON student_topic_progress(student_id, topic_id);
```
